00075 void task_motor::run (void)
00076 {
00077     // Make a variable which will hold times to use for precise task scheduling
00078     portTickType previousTicks = xTaskGetTickCount ();
00079     
00080 
00081     // Configure counter/timer 3 as a PWM for motors. First set the data
00082     // direction register so that the pin used for the PWM will be an output. The 
00083     // pin is Port E pin 4, which is also OC3B (Output Compare B for Timer 3)
00084     clear_and_set( DDRE, 4);    //TODO: is there a name for DDRE bit 4 we can put here
00085                     //instead of the # 4? is OC3B defined?
00086     
00087 
00088     // TODO: Fish, is this stuff all needed for motors, or is some left over from the LED proj?
00089     //To set 8-bit fast PWM mode we must set bits WGM30 and WGM32, which are in two
00090     // different registers (ugh). We use COM3B1 and Com3B0 to set up the PWM so that
00091     // the pin output will have inverted sense, that is, a 0 is on and a 1 is off; 
00092     // this is needed because the LED connects from Vcc to the pin. 
00093     clear_and_set( TCCR3A, WGM30 );
00094     set_bit( TCCR3A, COM3B1 );
00095     set_bit( TCCR3A, COM3B0 );
00096     //TCCR3A =  (1 << WGM30) | (1 << COM3B1) | (1 << COM3B0);
00097 
00098     // The CS31 and CS30 bits set the prescaler for this timer/counter to run the
00099     // timer at F_CPU / 64
00100     clear_and_set( TCCR3B, WGM32 );
00101     set_bit( TCCR3B, CS31 );
00102     set_bit( TCCR3B, CS30 );
00103     //TCCR3B = (1 << WGM32) | (1 << CS31)  | (1 << CS30);
00104     
00105     
00106     //instantiate 2 motors.
00107         motor motor1(&DDRB, (uint8_t)PB6, &DDRC, (uint8_t)PC0, (uint8_t)PC1, (uint8_t)PC2, &PORTC, &OCR1B);
00108     motor motor2(&DDRB, (uint8_t)PB5, &DDRD, (uint8_t)PD5, (uint8_t)PD6, (uint8_t)PD7, &PORTD, &OCR1A);
00109     //motor1 is controlled outside of this task by "shared_motor1_state", using methods "get()" and "put()"
00110     //motor2 is controlled by "shared_motr2_state", the same way.
00111     
00112     
00113     for (;;)
00114     { 
00115             //if motor is commanded to run,...
00116                if ( shared_motor1_state.get() == RUN )
00117           //...then send the motor the commanded power
00118           motor1.set_power( shared_motor1_power.get() );
00119            else
00120           //...else it should be braking, so send the motor the brake command 
00121           motor1.brake( shared_motor1_braking.get() );
00122            
00123            //same as motor 1
00124            if ( shared_motor2_state.get() == RUN )
00125           motor2.set_power( shared_motor2_power.get() ); 
00126            else
00127           motor2.brake( shared_motor2_braking.get() );
00128         
00129 
00130         // This is a method we use to cause a task to make one run through its task
00131         // loop every N milliseconds and let other tasks run at other times
00132         delay_from_to (previousTicks, configMS_TO_TICKS (100));     
00133         
00134     }
