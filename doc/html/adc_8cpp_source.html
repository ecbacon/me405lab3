00056 /** \brief This constructor sets up an A/D converter. 
00057  *  \details The A/D is made ready so that when a method such as \c read_once() or 
00058  *  \c read_oversampled is called, correct A/D conversions can be performed. This is
00059  *      accomplished by enabling the ADC, setting the clock prescaler to "divide-by-32",
00060  *      and setting the analog reference voltage to AVCC with an external capacitor to 
00061  *      ground. Input should be a serial port. Successful construction is written to the
00062  *      debug.
00063  *  
00064  *  @param p_serial_port A pointer to the serial port which writes debugging info. 
00065  */
00066 
00067 adc::adc (emstream* p_serial_port)
00068 {
00069     ptr_to_serial = p_serial_port;
00070 
00071     ADCSRA = (1<<ADEN) | (1<<ADPS2) | (1<<ADPS0);// Set the "ADC Enable" bit within the ADC Status Register A,
00072                      // and set the ADC clock prescaler to divide by 32.
00073     
00074     ADMUX |= 1<<REFS0;// Set the analog reference voltage to AVCC.
00075     
00076         
00077     DBG (ptr_to_serial, "A/D constructor OK" << endl);

00082 /** \brief This method takes one A/D reading from the given channel and returns it. 
00083  *  \details If the counter reaches a maximum number of cycles of 256, adc will timeout
00084  *  @param  ch The A/D channel which is being read must be from 0 to 7
00085  *  @return The result of the A/D conversion
00086  */
00087 
00088 uint16_t adc::read_once (uint8_t ch)
00089 {
00090     ADMUX &= ~( (1<<MUX2)|(1<<MUX1)|(1<<MUX0) ); // Clear the ADC Multiplexor Selection bits, and the set them to  
00091     ADMUX |= ch & ( (1<<MUX2)|(1<<MUX1)|(1<<MUX0) ); // the appropriate ADC channel, "ch".
00092     
00093     ADCSRA |= 1<<ADSC;// Set the "ADC start conversion" bit, initiating a conversion.
00094     
00095     const uint16_t maxWaitCycles = 60000;
00096     uint16_t counter = 0;
00097     
00098     while(ADCSRA & (1<<ADSC) && maxWaitCycles > counter++)// Poll the ADC start conversion bit, which will remain
00099                                 // high until the conversion is complete.++++++++
00100       ;
00101     
00102     return ADC;// return the resulting value found in the ADC register.
00103 }

00107 /** \brief This method takes an average reading on an ADC
00108  *  \details Take an average reading from the adc that is on the channel input as a parameter.
00109  *      The number of samples to be averaged is determined by the parameter "samples". If the
00110  *      "samples" parameter exceeds the maximum permissible number of samples(63), then only
00111  *      63 samples are taken. The number of samples taken, and the oversampled value are 
00112  *      written to the debug stream. 
00113  *  @param channel
00114  *  @param samples
00115  *  @return adc Value
00116  */
00117 
00118 uint16_t adc::read_average (uint8_t channel, uint8_t samples)
00119 {
00120     uint16_t sum = 0;
00121     uint8_t count = 0;
00122     
00123     //throw away first value
00124     read_once(channel);
00125     
00126     for(count = 0; count<samples && count < MAX_SAMPLES ;count++)
00127     {   
00128         sum += read_once(channel);
00129     }
00130     //Average the taken readings. Keep the result in the sum variable.
00131     return sum/count;
